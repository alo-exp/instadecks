function freshTmpDir(tag) { return fs.mkdtempSync(path.join(os.tmpdir(), `${tag}-`)); }

// Harness: simulate one cycle the way SKILL.md prescribes.
async function simulateCycle(runDir, cycle, scriptedFindings, opts = {}) {
  // 1. interrupt check
  if (checkInterrupt(runDir)) {
    await appendLedger(runDir, { cycle, /* ... */ ended_via: 'interrupted' });
    return { interrupted: true };
  }
  // 2-5. write fake deck.pptx + slide JPGs (simulating runCreate + pptx-to-images.sh)
  const cycleDir = path.join(runDir, `cycle-${cycle}`);
  await fsp.mkdir(path.join(cycleDir, 'slides'), { recursive: true });
  for (let i = 1; i <= 3; i++) {
    const buf = opts.slidesIdenticalToPrevCycle
      ? Buffer.from(`slide-${i}-cycle-${cycle - 1}-bytes`)
      : Buffer.from(`slide-${i}-cycle-${cycle}-bytes`);
    await fsp.writeFile(path.join(cycleDir, 'slides', `slide-0${i}.jpg`), buf);
  }
  // 2. Decide review_mode per SKILL.md step 2 (mirrors D-07):
  //    - cycle === 1                                                 → 'full'
  //    - cycle === 2 AND prior ledger entry findings_genuine === 0   → 'full'  (D-07 confirmation)
  //    - else                                                        → 'diff-only'
  const priorLedger = await readLedger(runDir);
  const priorEntry = priorLedger[priorLedger.length - 1] || null;
  let review_mode;
  if (cycle === 1) {
    review_mode = 'full';
  } else if (cycle === 2 && priorEntry && priorEntry.findings_genuine === 0) {
    review_mode = 'full';      // D-07 forced confirmation cycle
  } else {
    review_mode = 'diff-only';
  }
  // 6. slidesToReview is null for full mode, else the SHA-diff list:
  const slidesToReview = review_mode === 'full'
    ? null
    : await slidesChangedSinceLastCycle(runDir, cycle);
  // 7. mocked runReview returns scriptedFindings entries
  const findings = scriptedFindings;
  // 8. triage (every finding genuine = true unless scripted otherwise)
  // 9. issue_set_hash
  const genuineSet = findings.slides.flatMap(s => s.findings.map(f => ({slideNum: s.slideNum, text: f.text})));
  const hash = hashIssueSet(genuineSet);
  // 10. appendLedger
  await appendLedger(runDir, {
    cycle, timestamp: new Date().toISOString(),
    findings_total: genuineSet.length,
    findings_genuine: genuineSet.length,
    findings_fixed: 0,
    findings_intentionally_skipped: 0,
    issue_set_hash: hash,
    skipped_finding_ids: [],
    fixed_finding_ids: [],
    slides_changed: slidesToReview || [],
    review_mode,
    ended_via: null,
  });
  // 11. detectOscillation check happens at caller scope.
  return { interrupted: false, hash, genuineCount: genuineSet.length, review_mode };
}

```js
// skills/create/scripts/cli.js — add near top
function isInteractive() {
  if (process.env.CI === '1' || process.env.CI === 'true') return false;
  if (process.env.NON_INTERACTIVE === '1') return false;
  if (!process.stdout.isTTY) return false;
  return true;
}

// Argv parsing: add --soft-cap=<accept|stop|continue>
function parseSoftCapFlag(argv) {
  const flag = argv.find(a => a.startsWith('--soft-cap='));
  if (!flag) return null;
  const val = flag.slice('--soft-cap='.length);
  if (!['accept', 'stop', 'continue'].includes(val)) {
    throw new Error(`--soft-cap must be one of: accept, stop, continue (got ${JSON.stringify(val)})`);
  }
  return val;
}

// At the soft-cap decision point (cycle 5 reached without convergence):
// - if isInteractive() → AskUserQuestion (handled by SKILL.md / agent path; CLI alone cannot ask)
// - else if --soft-cap flag set → use that value
// - else default to 'accept' with stderr warning
function resolveSoftCap(softCapFlag) {
  if (softCapFlag) return softCapFlag;
  if (!isInteractive()) {
    process.stderr.write(
      "Instadecks: cycle 5 reached without convergence; non-interactive mode → accepting current deck. " +
      "Use --soft-cap=stop or interactive run for choice.\n"
    );
    return 'accept';
  }
  // interactive standalone CLI path — for v0.1.0 we still default to accept with warning;
  // full AskUserQuestion is agent-mode only (D-05).
  process.stderr.write("Instadecks: cycle 5 reached without convergence; defaulting to accept. Use --soft-cap=stop to override.\n");
  return 'accept';
}

module.exports = { isInteractive, parseSoftCapFlag, resolveSoftCap };
```
