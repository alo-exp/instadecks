**Ledger entry schema (D-02):**
```js
{
  cycle: number,                           // positive int, 1-indexed
  timestamp: string,                       // ISO 8601
  findings_total: number,
  findings_genuine: number,
  findings_fixed: number,
  findings_intentionally_skipped: number,
  issue_set_hash: string,                  // 'sha1:' + 40-hex
  skipped_finding_ids: string[],
  fixed_finding_ids: string[],
  slides_changed: number[],
  review_mode: 'full' | 'diff-only',
  ended_via: null | 'converged' | 'oscillation' | 'soft-cap-accepted' | 'soft-cap-stopped' | 'interrupted'
}
```

**lib/loop-primitives.js exports:**
```js
async function appendLedger(runDir, entry)        // throws on missing/bad cycle field
async function readLedger(runDir)                 // [] on ENOENT; tolerates truncated final line
function checkInterrupt(runDir)                   // sync; fs.existsSync(path.join(runDir, '.interrupt'))
function hashIssueSet(findings)                   // findings: [{slideNum, text}, ...]; returns 'sha1:<40-hex>'
async function slideImagesSha(cycleDir)           // returns {[filename]: sha256-hex} for slide-NN.jpg matches
async function slidesChangedSinceLastCycle(runDir, cycle)  // null on cycle<=1; sorted int[] otherwise

module.exports = {
  appendLedger, readLedger, checkInterrupt,
  hashIssueSet, slideImagesSha, slidesChangedSinceLastCycle,
  _internal: { /* helpers */ },
};
```

Path layout:
- Ledger: `<runDir>/refine-ledger.jsonl`
- Cycle slides: `<runDir>/cycle-${N}/slides/slide-${NN}.jpg` (NN zero-padded 2 digits)
- Interrupt flag: `<runDir>/.interrupt`

`hashIssueSet` algorithm: map each finding to `${slideNum}|${text.trim()}`; sort lexicographically; join with `\n`; SHA-1 the result; prefix `sha1:`. Reorder-invariant; whitespace trimmed.

`slidesChangedSinceLastCycle(runDir, cycle)`:
- If `cycle <= 1` → return `null` (sentinel for "all").
- Read `<runDir>/cycle-${cycle}/slides/` and `<runDir>/cycle-${cycle-1}/slides/`.
- For each `slide-NN.jpg` present in current cycle dir, compare SHA-256 against prior cycle's same filename.
- If filename absent in prior cycle → treat as changed.
- Returns sorted int[] of NN values.

**lib/oscillation.js — detectOscillation(ledger):**
```js
function detectOscillation(ledger) {
  if (!Array.isArray(ledger)) throw new Error('detectOscillation: ledger must be array');
  if (ledger.length < 3) return false;
  const N = ledger[ledger.length - 1];
  const Nm2 = ledger[ledger.length - 3];
  if (!N || !Nm2) return false;
  if ((N.findings_genuine || 0) === 0) return false;     // no oscillation if converged
  return Boolean(N.issue_set_hash && N.issue_set_hash === Nm2.issue_set_hash);
}
module.exports = { detectOscillation };
```
D-09 / Pitfall 2 refinement: strict hash equality on the unfixed-genuine set across N and N-2. Pure function. No fs.
