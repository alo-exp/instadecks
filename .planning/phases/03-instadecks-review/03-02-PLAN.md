**resolveSiblingOutputs (matches Phase 2 precedent — sibling-of-input):**
```js
function resolveSiblingOutputs(deckPath) {
  const dir = path.dirname(deckPath);
  const ext = path.extname(deckPath);
  const base = path.basename(deckPath, ext);
  return {
    jsonPath: path.join(dir, `${base}.review.json`),
    mdPath:   path.join(dir, `${base}.review.md`),
    narrativePath: path.join(dir, `${base}.review.narrative.md`),
  };
}
```

    'use strict';
    // cli.js — Standalone CLI for /instadecks:review (RVW-07).
    // Usage: node ${CLAUDE_PLUGIN_ROOT}/skills/review/scripts/cli.js <deckPath> [--findings <path>] [--run-id <id>] [--out-dir <dir>] [--annotate]
    // Reads findings JSON from --findings if supplied; calls runReview({mode:'standalone'}); exits 0 on success.

    const fs = require('node:fs');
    const { runReview } = require('./index');

    function parseArgs(argv) { /* simple parse: positional deckPath, then --flags */ }

    async function main() {
      const args = parseArgs(process.argv.slice(2));
      if (!args.deckPath) { console.error('Usage: cli.js <deckPath> [--findings <path>] [--run-id <id>] [--out-dir <dir>] [--annotate]'); process.exit(1); }
      let findings;

    }
    main().catch(e => { console.error(e.message); process.exit(3); });
